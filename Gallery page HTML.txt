   <!-- jquery for carousel slide when user click the next or previous btn -->
   <script>
    // Select the carousel container and the cards
    const carouselSlide = document.getElementById('carouselSlide');
    const cards = document.querySelectorAll('.gallery-card-item'); // All the card elements
    const totalCards = cards.length;
    let cardsPerSlide = 3; // Default to 3 cards per slide
    let currentIndex = 0; // Starting index of the cards to display

    // Function to update the visible cards
    function updateCarousel() {
        // Hide all cards initially
        cards.forEach(card => card.style.display = 'none');

        // Determine the cards that should be shown based on the current index
        const visibleCards = Array.from(cards).slice(currentIndex, currentIndex + cardsPerSlide);

        // Show the selected set of cards
        visibleCards.forEach(card => card.style.display = 'block');
    }

    // Next button event listener
    document.getElementById('nextBtn').addEventListener('click', function() {
        if (currentIndex + cardsPerSlide < totalCards) {
            currentIndex += cardsPerSlide; // Move to the next set of cards
        } else {
            currentIndex = 0; // Reset to the first set of cards
        }
        updateCarousel(); // Update the carousel display
    });

    // Previous button event listener
    document.getElementById('prevBtn').addEventListener('click', function() {
        if (currentIndex - cardsPerSlide >= 0) {
            currentIndex -= cardsPerSlide; // Move to the previous set of cards
        } else {
            currentIndex = totalCards - cardsPerSlide; // Move to the last set of cards
        }
        updateCarousel(); // Update the carousel display
    });

    // Adjust the number of cards per slide based on the screen size
    function adjustCarouselLayout() {
        if (window.innerWidth >= 1200) {
            cardsPerSlide = 3;
        } else if (window.innerWidth >= 768) {
            cardsPerSlide = 2;
        } else {
            cardsPerSlide = 1;
        }
        updateCarousel(); // Recalculate the cards to show based on the new layout
    }

    // Listen for window resize events to adjust the carousel layout dynamically
    window.addEventListener('resize', adjustCarouselLayout);

    // Initial load of the carousel with the first set of cards
    adjustCarouselLayout();
</script>
